00028  Function:              calc_least_sq_fit
00029  Last Modified Date:    25/01/11
00030  Purpose:               finds the coefficients of the [order] fit, for
00031                         [equations], whose values are defined in [array_x]
00032                         and [array_y]. stores in [coeff] where coeff[0] 
00033                         is the 0th order of fit with a corresponding 
00034                         [chi_squared]
00035  Required By:           frodo_red_findpeaks_simple.c
00036                         frodo_red_arcfit.c
00037  Additional Notes:
00038 
00039 
00040  Performs least-squares fitting to a general model y = Xc using the GSL
00041  library.
00042 
00043                         Array formats
00044                         -------------
00045 
00046  where m is the order and n is the number of equations.
00047 
00048  y = [  y_0, y_1, y_2, ..., y_n;        ]
00049 
00050  X = [  1, (x_0)^1, (x_0)^2, ..., (x_0)^m;
00051         1, (x_1)^1, (x_1)^2, ..., (x_1)^m;
00052         1, (x_2)^1, (x_2)^2, ..., (x_2)^m;
00053         1, (x_3)^1, (x_3)^2, ..., (x_3)^m;
00054         .       .       .       .       .
00055         .       .       .       .       .
00056         .       .       .       .       .
00057         1, (x_n)^1, (x_n)^2, ..., (x_n)^m; ]
00058 
00059  c = [  c_0, c_1, c_2, ..., c_n+1;      ]
00060 
00061  n.b. matrices are are denoted in (rows x columns) format.
00062 
00063 ************************************************************************/

00065 int calc_least_sq_fit(int order, int equations, double array_x [], double array_y [], double coeffs [], double *chi_squared) {
00066 
00067         if (order >= equations) {       // then we don't have enough equations to solve for all parameters
00068 
00069                 return 1;
00070 
00071         }
00072 
00073         int ii, jj;
00074 
00075         gsl_matrix *X, *cov;
00076         gsl_vector *y, *c;
00077 
00078         double chisq;
00079 
00080         X = gsl_matrix_alloc(equations, order+1);       // matrix of predictor variables
00081         y = gsl_vector_alloc(equations);                // vector of observed values
00082 
00083         c = gsl_vector_alloc(order+1);                  // coefficient vector
00084         cov = gsl_matrix_alloc(order+1, order+1);       // covariance matrix
00085 
00086         for (ii=0; ii<equations; ii++) {                                        // for each equation
00087                         
00088                 for (jj=0; jj<order+1; jj++) {
00089                            
00090                         gsl_matrix_set(X, ii, jj, powf(array_x[ii], jj));       // populate X matrix with the value of x^m for each order m.
00091 
00092                 }
00093                            
00094                 gsl_vector_set(y, ii, array_y[ii]);                             // and populate y matrix
00095 
00096         }
00097 
00098         gsl_multifit_linear_workspace * work = gsl_multifit_linear_alloc(equations, order+1);   // define a workspace
00099         gsl_multifit_linear(X, y, c, cov, &chisq, work);                                        // perform the fit
00100         gsl_multifit_linear_free(work);                                                         // free the workspace
00101 
00102         #define C(i) (gsl_vector_get(c,(i)))
00103 
00104         for (ii=0; ii<order+1; ii++) {
00105 
00106                 coeffs[ii] = C(ii);     // populate coeff array
00107                 
00108                 // printf("%e\t", coeffs[ii]);  if (ii==order) { printf("\n"); }        // DEBUG
00109 
00110         }
00111 
00112         *chi_squared = chisq;           // write chisq
00113              
00114         gsl_matrix_free(X);
00115         gsl_vector_free(y);
00116         gsl_vector_free(c);
00117         gsl_matrix_free(cov);
00118              
00119         return 0;
00120 
00121 }

00125  Function:              calculate_cross_correlation_offset
00126  Last Modified Date:    17/01/11
00127  Purpose:               calculates the best offset for two given
00128                         datasets, x and y
00129  Required By:           frodo_red_arcfit.c
00130  Additional Notes:      
00131 
00132  The cross correlation coefficient, r, is determined by the equation:
00133 
00134  r =
00135 
00136                 sum_i[(x(i) - mean_x) * y(i-offset) - mean_y)]
00137  ------------------------------------------------------------------------
00138  sqrt_(sum_i((x(i) - mean_x)^2)) * sqrt_(sum_i((y(i-offset) - mean_y)^2)) 
00139 
00140  a +ve offset indicates the series lags the reference series
00141  a -ve offset indicates the series leads the references series          
00142 
00143 ************************************************************************/
00144 
00145 int calculate_cross_correlation_offset(double x [], double y [], int n, int max_offset, int *this_best_offset, double *this_best_r) {
00146 
00147         double mean_x = gsl_stats_mean(x, 1, n);
00148         double mean_y = gsl_stats_mean(y, 1, n);
00149 
00150         int offset, best_offset;
00151         double r, best_r;
00152 
00153         for (offset=-max_offset; offset<=max_offset; offset++) {
00154         
00155                 double sx, sy, numerator = 0.0, den_x_term = 0.0, den_y_term = 0.0, denominator;
00156 
00157                 int ii, jj;
00158         
00159                 for (ii=0; ii<n; ii++) {
00160                 
00161                         jj = ii + offset;
00162 
00163                         if (jj < 0 || jj >= n) {        // element is outside of array boundaries
00164 
00165                                 continue;
00166 
00167                         } else {
00168 
00169                                 sx = (x[ii] - mean_x);
00170                                 sy = (y[jj] - mean_y);
00171 
00172                                 numerator += sx * sy;
00173                                 den_x_term += powf(sx, 2);
00174                                 den_y_term += powf(sy, 2);
00175                         
00176                         }
00177 
00178                         denominator = sqrt(den_x_term) * sqrt(den_y_term);
00179         
00180                 }
00181 
00182                 r = numerator / denominator;
00183 
00184                 if ((offset == -max_offset) || (r > best_r)) {  // Comparing doubles but accuracy isn't a necessity so don't need gsl_fcmp function
00185 
00186                         best_r = r;
00187                         best_offset = offset;
00188 
00189                 } 
00190 
00191                 // printf("%d\t%f\t%f\t%f\n", offset, numerator/denominator, numerator, denominator);   // DEBUG
00192 
00193         }
00194 
00195         *this_best_offset = best_offset;
00196         *this_best_r = best_r;
00197 
00198         // printf("%d\t%f\n", best_offset, best_r);             // DEBUG
00199 
00200         return 0;
00201 
00202 }

00672  Function:              iterative_sigma_clip
00673  Last Modified Date:    08/03/11        
00674  Purpose:               Perform iterative sigma clipping on a dataset of 
00675                         n values
00676  Required By:           frodo_red_subsky
00677  Additional Notes:      
00678 
00679  Before each iteration, a ceiling value is worked out which is found by:
00680 
00681   ceiling = [mean] + [clip_sigma] * [sd]
00682 
00683  The elements in the dataset [values] are then cycled and any value that
00684  is greater than the ceiling value for this iteration are flagged FALSE
00685  in the [retain_indexes] array. This procedure is recursively done until
00686  the number of indexes retained for an iteration equals that of the last
00687  iteration.
00688 
00689  Along with the [retain_indexes] array, the final mean [final_mean],
00690  final standard deviation [final_sd] and final number of retained 
00691  indexes [final_num_retained_indexes] variables are populated.
00692 
00693 ************************************************************************/
00694 
00695 int iterative_sigma_clip(double values [], int n, int clip_sigma, int retain_indexes [], double * final_mean, double * final_sd, int * final_num_retained_indexes) {
00696 
00697         int ii;
00698 
00699         int this_iteration_num_retained_indexes, last_iteration_num_retained_indexes = n;
00700 
00701         int this_iteration_values_index;
00702 
00703         int iteration_count = 0;
00704 
00705         bool continue_iterations = TRUE;
00706 
00707         for (ii=0; ii<n; ii++) {        // initialise to TRUE under assumption that all are to be retained
00708 
00709                 retain_indexes[ii] = TRUE;
00710 
00711         }
00712 
00713         double mean = gsl_stats_mean(values, 1, n);
00714         double sd = gsl_stats_sd(values, 1, n);
00715 
00716         double ceiling = mean + (double) clip_sigma*sd;
00717 
00718         while (continue_iterations == TRUE) {
00719 
00720                 this_iteration_num_retained_indexes = n;
00721 
00722                 // ***********************************************************************
00723                 // Determine which indexes to retain given current ceiling value
00724 
00725                 for (ii=0; ii<n; ii++) {
00726 
00727                         if (values[ii] > ceiling) {     // this index, ii, is not to be retained
00728         
00729                                 retain_indexes[ii] = FALSE;
00730                                 this_iteration_num_retained_indexes--;
00731 
00732                         }
00733 
00734                 }
00735 
00736                 //      printf("%f\t%f\t%f\n", mean, sd, ceiling);      // DEBUG
00737 
00738                 // ***********************************************************************
00739                 // Check to see if the number of retained indexes is the same as the
00740                 // previous iteration. If this is the first iteration, check to see if the
00741                 // number of retained indexes is equal to the original number of values.
00742                 // If so, stop the iterations.
00743 
00744                 if ((this_iteration_num_retained_indexes == last_iteration_num_retained_indexes) || this_iteration_num_retained_indexes <= 1) {
00745 
00746                         break;
00747 
00748                 }
00749 
00750                 iteration_count++;      
00751 
00752                 // ***********************************************************************
00753                 // Create new arrays of retained values and work out new ceiling value
00754 
00755                 double *this_iteration_values;
00756                 this_iteration_values = (double *) malloc((this_iteration_num_retained_indexes)*sizeof(double));
00757 
00758                 this_iteration_values_index = 0;
00759 
00760                 for (ii=0; ii<n; ii++) {
00761 
00762                         if (retain_indexes[ii] == TRUE) {
00763 
00764                                 this_iteration_values[this_iteration_values_index] = values[ii];
00765                                 this_iteration_values_index++;
00766 
00767                         }
00768 
00769                 }
00770 
00771                 printf("\nIteration:\t\t\t%d\n", iteration_count);
00772                 printf("Ceiling:\t\t\t%.3e\n", ceiling);
00773                 printf("Number of indexes retained:\t%d\n", this_iteration_num_retained_indexes);
00774 
00775                 mean = gsl_stats_mean(this_iteration_values, 1, this_iteration_values_index);
00776                 sd = gsl_stats_sd(this_iteration_values, 1, this_iteration_values_index);
00777 
00778                 ceiling = mean + (double) clip_sigma*sd;
00779 
00780                 last_iteration_num_retained_indexes = this_iteration_num_retained_indexes;
00781         
00782                 free(this_iteration_values);
00783                         
00784         }
00785 
00786         *final_mean = mean;
00787         *final_sd = sd;
00788 
00789         *final_num_retained_indexes = this_iteration_num_retained_indexes;
00790 
00791         return 0;
00792 
00793 }
