00285 void getdat (int *ierr)
00286 /*----------------------------------------------------------------------- */
00287 /*   Read FITS file and determine max. and min. values */
00288 /*   Inputs in common: */
00289 /*      fptr    Input FITS fitsio unit number */
00290 /*   Output: */
00291 /*      ierr  I    Error code: 0 => ok */
00292 /*   Output in common: */
00293 /*      vmax    Maximum image value */
00294 /*      vmin    Minimum image value */

00296 /*----------------------------------------------------------------------- */

00298   long size,incs[7]={1,1,1,1,1,1,1},blc[7]={1,1,1,1,1,1,1},trc[7];
00299   int group=0, i, anyf;
00300   float tmax,tmin;
00301 
00302 /* How big is the array? */
00303   size = inaxes[0];
00304   size = size * inaxes[1];
00305 /* allocate floating array */
00306   if (DataArray) free(DataArray); /* free any old allocations */
00307   DataArray = (float*)malloc(sizeof(float)*size);
00308   if (!DataArray) { /* cannot allocate */
00309     fprintf(stderr,"Cannot allocate memory for image array \n");
00310     *ierr = 1;
00311     return;
00312   }
00313 /*                                       Take all of image */
00314   trc[0] = inaxes[0];
00315   trc[1] = inaxes[1];
00316 /*                                       but only first plane */
00317   for (i=2;i<naxis;i++) trc[i] = 1;
00318 /*                                       Read selected portion of input */
00319   fits_read_subset_flt (fptr, group, naxis, inaxes, blc, trc, incs,
00320         fblank, DataArray, &anyf, ierr);
00321   if (*ierr!=0) {
00322     fprintf(stderr,"fits_read_subset_flt error reading input file \n");
00323     return;
00324   }
00325 /*                                      Update data max, min */
00326   tmin = 1.0e20;
00327   tmax = -1.0e20;
00328   for (i=0;i<size;i++) {
00329     if ((!anyf) || (DataArray[i]!=fblank)) {
00330       if (DataArray[i]>tmax) tmax = DataArray[i];
00331       if (DataArray[i]<tmin) tmin = DataArray[i];
00332     }
00333   }
00334   if (GotMaxMin) {
00335 /*                                          Don't put vmax,vmin outside */
00336 /*                                          of actual range */
00337     if (vmin<tmin) vmin = tmin;
00338     if (vmax>tmax) vmax = tmax;
00339   }
00340   else { /* set to full range */
00341     GotMaxMin = 1;
00342     vmax = tmax;
00343     vmin = tmin;
00344   }
