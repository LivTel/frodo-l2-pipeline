00043 void jpgini (char *iname, int inx, int iny, float ivmax,  float ivmin, 
00044              int nonLin, int quality, int *ierr)
00045 /*--------------------------------------------------------------------*/
00046 /*  Initializes i/o to jpeg output routines                           */
00047 /*  Inputs:                                                           */
00048 /*     iname    Name of output file                                   */
00049 /*     ilname   Length of name                                        */
00050 /*     inx      number of columns in image                            */
00051 /*     iny      Number of rows in image                               */
00052 /*     ivmax    Maximum image value (values larger get this value)    */
00053 /*     ivmin    Minimum image value (values smaller get this value)   */
00054 /*     nonLin   if > 0.0 use non linear function                      */
00055 /*     quality  jpeg quality factor [1-100]                           */
00056 /*  Output:                                                           */
00057 /*     ierr     0.0 => OK                                             */
00058 /*--------------------------------------------------------------------*/
00059 {
00060   int lname, i;
00061 
00062   /* get values */
00063   nx = inx;
00064   ny = iny;
00065   lname = strlen(iname);
00066   vmax = ivmax;
00067   vmin = ivmin;
00068   nonlinear = nonLin;
00069 
00070   /* file name */
00071   if (name) free(name);
00072   name = (char*)malloc(lname+1);
00073   if (name == NULL) {
00074     fprintf(stderr, "can't allocate file name");
00075     *ierr = 1; /* set error return */
00076     return; /* failed */
00077   }
00078   for (i=0;i<lname;i++) name[i] = iname[i]; name[i] = 0;
00079 
00080   /* row buffer */
00081   if (idata) free(idata);
00082   idata = (JSAMPROW)malloc(nx); /* allocate for gray scale */
00083   if (idata == NULL) {
00084     fprintf(stderr, "can't allocate row buffer");
00085     *ierr = 2; /* set error return */
00086     return; /* failed */
00087   }
00088   for (i=0;i<nx;i++) idata[i] = 0;
00089 
00090   /* open output file */
00091   if ((outfile = fopen(name, "wb")) == NULL) {

00096 
00097   /* create jpeg structures */

00099   jpeg_create_compress(&cinfo);

00123 void jpgwri (float *data, float blank, int *ierr)
00124 /*--------------------------------------------------------------------*/
00125 /*  Write row of a jpeg image, floating values scaled and converted   */
00126 /*  as specified to jpgini.  Grayscale only.                          */
00127 /*  Only does gray scale at present                                   */
00128 /*  Pure black reserved for blanked pixels.                           */
00129 /*  Inputs:                                                           */
00130 /*     data    floating values                                        */
00131 /*     blank   value of undefined pixel                               */
00132 /*  Output:                                                           */
00133 /*     ierr     0.0 => OK                                             */
00134 /*--------------------------------------------------------------------*/
00135 {
00136   int i, icol, maxcolor=255;
00137   float val, irange, c1, c2;
00138   double arg;
00139 
00140   /* scaling parameters */
00141    irange = vmax - vmin;
00142    if (fabs(irange)<1.0e-25)
00143           irange = 1.0;
00144    else
00145       irange = 1.0 / irange;
00146    c1 = (maxcolor - 1.0) * irange;
00147    c2 = vmin * c1 - 0.5;
00148 
00149   /* convert row */
00150   for (i=0;i<nx;i++) {
00151     val = data[i];
00152     if (val==blank) { /* blanked */
00153       idata[i] = 0;
00154     }
00155     else
00156       { /* non blanked */
00157         if (val<vmin) val=vmin;
00158         if (val>vmax) val=vmax;
00159         if (nonlinear)  {/* nonlinear */
00160           arg = ((val-vmin) * irange);
00161           icol =  0.5 + ((maxcolor-1.0) * sqrt(arg));
00162         }
00163         else  /* Linear */
00164           icol = c1 * val - c2;
00165         if (icol<1) icol = 1;  /* minimum color = 1 */
00166         if (icol>=maxcolor) icol = maxcolor-1;
00167         idata[i] = GETJSAMPLE(icol);
00168       }
00169   } /* end loop over row pixels */
00170 
00171   row_pointer[0] = (JSAMPROW)idata; /* set row pointer */
00172 
00173   /* compress/write row */
00174   jpeg_write_scanlines(&cinfo, row_pointer, 1);
00175 
00176   *ierr = 0; /* OK */
00177 } /* end of jpgwri */
